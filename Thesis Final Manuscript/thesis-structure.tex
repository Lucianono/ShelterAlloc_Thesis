	{\protect\numberline{\thechapter}\texorpdfstring{\uppercase{#1}}{#1}}%

	\addtocontents{lot}{\protect\addvspace{10\p@}}%
	\if@twocolumn
	\@topnewpage[\@makechapterhead{#2}]%
	\else
	\@makechapterhead{#2}%
	\@afterheading
	\fi}
